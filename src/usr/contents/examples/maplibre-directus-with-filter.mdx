### Maplibre: Adding a vector layer from a Directus API with filter

Only Hellenistic and Roman sites are being displayed

<div className="row mt-3 mb-5 border-top border-bottom py-3">
  <div className="col-6">
    <MapCompLibre 
      center="37.00416,37.48551,9"
      baseLayers={["CAWM", "EsriSatellite"]}
    >
      <VectorLayerLibre
        name="Hellenistic and Roman Sites"
        source={{
          dTable: "scms_ksa",
          geoField: "geometry",
          dQueryString: "filter[Hellenistic_Roman][_eq]=true"  
        }}
        checked={true}
        fitToContent={true}
        type="geojson"

          "Item_Label": "Label",
